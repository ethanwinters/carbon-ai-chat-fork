## Overview

`cds-aichat-reasoning-steps-toggle` is a dedicated control for expanding or collapsing a `cds-aichat-reasoning-steps` wrapper. Use it when the host application owns the open state instead of relying on custom markup.

### Getting started

```html
<script type="module">
  const toggle = document.querySelector("cds-aichat-reasoning-steps-toggle");
  const wrapper = document.querySelector("cds-aichat-reasoning-steps");

  toggle.addEventListener("reasoning-steps-toggle", (event) => {
    wrapper.open = event.detail?.open;
  });
</script>

<cds-aichat-reasoning-steps-toggle
  panel-id="reasoning-toggle-panel"
  open-label-text="Hide reasoning steps"
  closed-label-text="Show reasoning steps"
></cds-aichat-reasoning-steps-toggle>

<cds-aichat-reasoning-steps id="reasoning-toggle-panel" open>
  <cds-aichat-reasoning-step title="Collect context" open>
    Combined prompt, chat history, and system instructions.
  </cds-aichat-reasoning-step>
  <cds-aichat-reasoning-step title="Draft plan">
    Outlined the answer and validated it against requirements.
  </cds-aichat-reasoning-step>
</cds-aichat-reasoning-steps>
```
