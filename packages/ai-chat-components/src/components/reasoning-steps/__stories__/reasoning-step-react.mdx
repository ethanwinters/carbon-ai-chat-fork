## Overview

The `<ReasoningStep>` component wraps the `cds-aichat-reasoning-step` web component so you can show individual steps in React without wiring custom events yourself.

### Getting started

```jsx
import ReasoningSteps from "@carbon/ai-chat-components/es/react/reasoning-steps.js";
import ReasoningStep from "@carbon/ai-chat-components/es/react/reasoning-step.js";
import Markdown from "@carbon/ai-chat-components/es/react/markdown.js";

function ReasoningStepExample() {
  return (
    <ReasoningSteps open>
      <ReasoningStep title="Review retrieved context" open>
        <Markdown>
          Confirmed the retrieved snippets map to the user's request and
          captured any open questions before drafting a response.
        </Markdown>
      </ReasoningStep>
    </ReasoningSteps>
  );
}
```

### Handling toggle events

```jsx
function ControlledStep({ step }) {
  const [isOpen, setIsOpen] = React.useState(step.open);

  return (
    <ReasoningSteps open>
      <ReasoningStep
        title={step.title}
        open={isOpen}
        controlled
        onToggle={(event) => setIsOpen(event.detail.open)}
      >
        {step.body}
      </ReasoningStep>
    </ReasoningSteps>
  );
}
```
