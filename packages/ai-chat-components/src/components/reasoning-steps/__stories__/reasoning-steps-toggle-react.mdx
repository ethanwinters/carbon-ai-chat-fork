## Overview

`<ReasoningStepsToggle>` wraps the `cds-aichat-reasoning-steps-toggle` web component for React apps. Combine it with `<ReasoningSteps>` to drive wrapper visibility without hand-rolling a button.

### Getting started

```jsx
import ReasoningStepsToggle from "@carbon/ai-chat-components/es/react/reasoning-steps-toggle.js";
import ReasoningSteps from "@carbon/ai-chat-components/es/react/reasoning-steps.js";
import ReasoningStep from "@carbon/ai-chat-components/es/react/reasoning-step.js";

function ReasoningToggleExample() {
  const [open, setOpen] = React.useState(true);

  return (
    <>
      <ReasoningStepsToggle
        open={open}
        panelID="reasoning-toggle-react"
        openLabelText="Hide reasoning steps"
        closedLabelText="Show reasoning steps"
        onToggle={(event) => setOpen(event.detail?.open ?? !open)}
      />
      <ReasoningSteps open={open} id="reasoning-toggle-react">
        <ReasoningStep title="Gather facts" open>
          Combined prompt, history, and retrieved citations.
        </ReasoningStep>
        <ReasoningStep title="Draft answer">
          Structured response using verified facts.
        </ReasoningStep>
      </ReasoningSteps>
    </>
  );
}
```
