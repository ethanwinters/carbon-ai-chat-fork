# AI Chat History


## Getting started

Here's a quick example to get you started.

### JS (via import)

```javascript
import "@carbon/ai-chat-components/es/react/history.js";
```

### HTML

```jsx
import {
  HistoryShell,
  HistoryHeader,
  HistoryToolbar,
  HistoryContent,
  HistoryLoading,
  HistoryPanel,
  HistoryPanelMenu,
  HistoryPanelItem,
  HistoryPanelItems,
  HistorySearchItem,
  HistoryDeletePanel,
} from "@carbon/ai-chat-components/es/react/history";
import { PinFilled, Search } from "@carbon/icons-react";

function ChatHistory() {
  const [searchValue, setSearchValue] = useState("");

  const handleSearchInput = (event) => {
    setSearchValue(event.detail.value);
  };

  const handleHistoryItemAction = (event) => {
    if (event.detail.action === "Rename") {
      const element = event.detail.element;
      if (element) {
        element.rename = true;
      }
    }
  };

  return (
    <HistoryShell>
      <HistoryHeader headerTitle="Chats" />
      <HistoryToolbar onSearchInput={handleSearchInput} />
      <HistoryContent resultsLabel="Results" resultsCount="10">
        <HistoryPanel aria-label="Chat history">
          <HistoryPanelItems>
            <HistoryPanelMenu expanded title="Pinned">
              <PinFilled slot="title-icon" />
              <HistoryPanelItem
                id="chat-1"
                name="Important conversation"
                selected={true}
                actions={pinnedHistoryItemActions}
                onMenuAction={handleHistoryItemAction}
              />
            </HistoryPanelMenu>
            <HistoryPanelMenu expanded title="Today">
              <Search slot="title-icon" />
              <HistoryPanelItem
                id="chat-2"
                name="Recent chat"
                actions={historyItemActions}
                onMenuAction={handleHistoryItemAction}
              />
            </HistoryPanelMenu>
          </HistoryPanelItems>
        </HistoryPanel>
      </HistoryContent>
    </HistoryShell>
  );
}
```

### Delete confirmation panel

Render **`HistoryDeletePanel`** when the user chooses **Delete** from a chat item’s overflow menu. Pass **`itemId`** for the chat item being removed.
Combined with **`HistoryShell`**, focus moves to the next chat item after delete, and **`history-item-selected`** fires on that chat item when the deleted chat item was the active one.

Use **`onCancel`** to dismiss the panel without deleting, and **`onConfirm`** to remove the chat from your state and hide the panel.

```jsx
{showDeletePanel && (
  <HistoryDeletePanel
    itemId={itemToDelete ?? ""}
    onCancel={handleDeleteCancel}
    onConfirm={handleDeleteConfirm}
  >
    <div slot="title">Confirm Delete</div>
    <div slot="description">
      This conversation will be permanently deleted.
    </div>
  </HistoryDeletePanel>
)}
```

## Custom Events

The chat history React components emit several custom events that you can listen for to handle user interactions. Below are tables documenting all available events for each component:

    <tr>
      <td><code>onRenameSave</code></td>
      <td><code>history-panel-item-input-save</code></td>

      <td>
        • <code>newName</code> (string) - The new name entered<br/>
        • <code>itemId</code> (string) - The ID of the item being renamed
      </td>
    </tr>
    <tr>
      <td><code>onRenameCancel</code></td>
      <td><code>history-panel-item-input-cancel</code></td>
      <td>Bubbles up from the inner input when the user cancels renaming (cancel button, Escape key, or focus-out with no valid change).</td>
      <td>None</td>
    </tr>
  </tbody>
</table>

**Example:**
```jsx
function ChatHistory() {
  const [items, setItems] = useState(chatItems);

  const handleItemSelected = (event) => {
    const { itemId, itemName } = event.detail;
    console.log(`Selected chat: ${itemName} (ID: ${itemId})`);
    // Load the selected conversation
    loadConversation(itemId);
  };

  const handleItemAction = (event) => {
    const { action, itemId, itemName, element } = event.detail;

    switch(action) {
      case 'Rename':
        element.rename = true;
        break;
      case 'Delete':
        console.log(`Deleting chat: ${itemName}`);
        deleteChat(itemId);
        break;
      case 'Pin':
        console.log(`Pinning chat: ${itemName}`);
        pinChat(itemId);
        break;
    }
  };

  const handleRenameChange = (event) => {
    const { value, itemId } = event.detail;
    const invalid = value.length > 75;
    setItems(prev => prev.map(item =>
      item.id === itemId
        ? { ...item, renameInvalid: invalid, renameInvalidMessage: invalid ? 'Title cannot exceed 75 characters.' : '' }
        : item
    ));
  };

  const handleRenameSave = (event) => {
    const { newName, itemId } = event.detail;
    setItems(prev => prev.map(item =>
      item.id === itemId ? { ...item, name: newName } : item
    ));
  };

  return (
    <HistoryPanelItems>
      {items.map(item => (
        <HistoryPanelItem
          key={item.id}
          id={item.id}
          name={item.name}
          rename={item.rename}
          renameInvalid={item.renameInvalid}
          renameInvalidMessage={item.renameInvalidMessage ?? ''}
          actions={historyItemActions}
          onSelected={handleItemSelected}
          onMenuAction={handleItemAction}
          onRenameChange={handleRenameChange}
          onRenameSave={handleRenameSave}
        />
      ))}
    </HistoryPanelItems>
  );
}
```

#### Focus Management Utility

When performing actions that reorder or move history items (such as pinning/unpinning), you may need to restore focus to the item after the DOM updates for accessibility reasons. Use the `focusElementAfterRepaint` utility to handle this:

```javascript
import { focusElementAfterRepaint } from "@carbon/ai-chat-components/es/globals/utils/focus-utils";
```

**Example with Pin/Unpin:**
```jsx
import { focusElementAfterRepaint } from "@carbon/ai-chat-components/es/globals/utils/focus-utils";

function ChatHistory() {
  const historyShellRef = useRef(null);

  const handlePin = (itemId) => {
    // Move item from regular to pinned list
    const itemToPin = regularItems.find(chat => chat.id === itemId);
    setPinnedItems(prev => [itemToPin, ...prev]);
    setRegularItems(prev => prev.filter(chat => chat.id !== itemId));

    // Restore focus to the item after DOM updates
    focusElementAfterRepaint(
      historyShellRef.current ?? document,
      `cds-aichat-history-panel-item[id="${CSS.escape(itemId)}"]`
    );
  };

  return (
    <HistoryShell ref={historyShellRef}>
      {/* ... history content */}
    </HistoryShell>
  );
}
```

```jsx
function RenameInput() {
  const [invalid, setInvalid] = useState(false);
  const [invalidMessage, setInvalidMessage] = useState('');

  const handleChange = (event) => {
    const { value } = event.detail;
    const isInvalid = value.length > 75;
    setInvalid(isInvalid);
    setInvalidMessage(isInvalid ? 'Title cannot exceed 75 characters.' : '');
  };

  const handleSave = (event) => {
    const { newName, itemId } = event.detail;
    console.log(`Item ${itemId} renamed to: ${newName}`);
    // Update the item name in your data store
    updateChatName(itemId, newName);
  };

  const handleCancel = () => {
    console.log('Rename cancelled');
    // Nothing to do — the parent item resets its own rename state
  };

  return (
    <HistoryPanelItemInput
      value="Current title"
      itemId="chat-1"
      invalid={invalid}
      invalidMessage={invalidMessage}
      onChange={handleChange}
      onSave={handleSave}
      onCancel={handleCancel}
    />
  );
}
```

**Example:**
```jsx
function ChatHistory() {
  const [showDeletePanel, setShowDeletePanel] = useState(false);
  const [itemToDelete, setItemToDelete] = useState(null);

  const handleDeleteAction = (event) => {
    const { action, itemId } = event.detail;
    if (action === 'Delete') {
      setItemToDelete(itemId);
      setShowDeletePanel(true);
    }
  };

  const handleDeleteCancel = () => {
    console.log('Delete cancelled');
    setShowDeletePanel(false);
    setItemToDelete(null);
  };

  const handleDeleteConfirm = () => {
    console.log('Delete confirmed for chat:', itemToDelete);
    deleteChat(itemToDelete);
    setShowDeletePanel(false);
    setItemToDelete(null);
  };

  return (
    <HistoryShell>
      <HistoryHeader headerTitle="Chats" />
      <HistoryToolbar />
      <HistoryContent>
        <HistoryPanel aria-label="Chat history">
          <HistoryPanelItems>
            <HistoryPanelMenu expanded title="Today">
              <HistoryPanelItem
                id="chat-1"
                name="My conversation"
                actions={historyItemActions}
                onMenuAction={handleDeleteAction}
              />
            </HistoryPanelMenu>
          </HistoryPanelItems>
        </HistoryPanel>

        {showDeletePanel && (
          <HistoryDeletePanel
            itemId={itemToDelete ?? ''}
            onCancel={handleDeleteCancel}
            onConfirm={handleDeleteConfirm}
          >
            <div slot="title">Confirm Delete</div>
            <div slot="description">
              This conversation will be permanently deleted.
            </div>
          </HistoryDeletePanel>
        )}
      </HistoryContent>
    </HistoryShell>
  );
}
```
