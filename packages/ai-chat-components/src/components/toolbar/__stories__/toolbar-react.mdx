## Overview

`<Toolbar>` aligns titles, labels, icon buttons, and overflow actions into a responsive header for AI chat surfaces. It handles truncation and optional overflow menus so you can focus on wiring your actions.

### Getting started

Here's a quick example to get you started.

```jsx
import Toolbar from "@carbon/ai-chat-components/es/react/toolbar.js";
import IconButton from "@carbon/ai-chat-components/es/react/icon-button.js";
import AILabel from "@carbon/ai-chat-components/es/react/ai-label.js";
import { Download, Launch } from "@carbon/icons-react";

function ToolbarExample() {
  return (
    <Toolbar overflow actions slot="action">
      <div slot="title" data-fixed>
        Conversation summary
      </div>
      <AILabel size="2xs" autoalign alignment="bottom" slot="toolbar-ai-label">
        <div slot="body-text">Powered by Watsonx</div>
      </AILabel>
      <IconButton kind="ghost" size="md" slot="action">
        <Download slot="icon" />
        <span slot="tooltip-content">Download</span>
      </IconButton>
      <IconButton kind="ghost" size="md" slot="action">
        <Launch slot="icon" />
        <span slot="tooltip-content">Launch</span>
      </IconButton>
    </Toolbar>
  );
}
```
