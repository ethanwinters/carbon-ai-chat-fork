## Overview

Use `cds-aichat-feedback-buttons` when you need quick thumbs up/down capture with optional detail panels. Pair it with `cds-aichat-feedback` to collect richer follow-up when users choose a sentiment.

### Getting started

Here's a quick example to get you started.

```html
<script type="module">
  const buttons = document.querySelector("cds-aichat-feedback-buttons");
  const panel = document.querySelector("cds-aichat-feedback");

  panel.categories = ["Inaccurate", "Unhelpful", "Inappropriate"];

  buttons.addEventListener("feedback-buttons-click", (event) => {
    const { isPositive } = event.detail;
    if (isPositive) {
      console.log("Positive feedback recorded");
    } else {
      panel.isOpen = true;
    }
  });

  panel.addEventListener("feedback-submit", (event) => {
    console.log("Detailed feedback:", event.detail);
    panel.isOpen = false;
  });
  panel.addEventListener("feedback-close", () => {
    panel.isOpen = false;
  });
</script>

<cds-aichat-feedback-buttons
  panel-id="feedback-panel-example"
  positive-label="I like this response"
  negativeLabel="I dislike this response"
  has-negative-details
></cds-aichat-feedback-buttons>

<cds-aichat-feedback
  id="feedback-panel-example"
  title="Additional feedback"
  prompt="Add a comment"
  text-area-placeholder="Add a comment"
  cancel-label="Close"
  submit-label="Submit"
  show-text-area
  show-prompt
></cds-aichat-feedback>
```
