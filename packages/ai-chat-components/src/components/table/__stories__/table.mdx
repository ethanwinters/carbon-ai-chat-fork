### Getting started

Here's a quick example to get you started.

```html
<script type="module">
  const table = document.querySelector("cds-aichat-table");
  table.headers = [{ text: "Name" }, { text: "Role" }, { text: "Status" }];
  table.rows = [
    {
      cells: [
        { text: "Jordan Smith" },
        { text: "Designer" },
        { text: "Active" },
      ],
    },
    {
      cells: [
        { text: "Priya Patel" },
        { text: "Engineer" },
        { text: "Paused" },
      ],
    },
    { cells: [{ text: "Lee Chen" }, { text: "PM" }, { text: "Active" }] },
  ];
</script>

<cds-aichat-table table-title="Agent roster"></cds-aichat-table>
```
